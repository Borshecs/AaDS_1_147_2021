from math import pi
from math import sqrt
def cycle():
    T = float(input(""))
    w = (2*pi)/T
    print(w)
cycle()
def period():
    m = float(input(""))
    k = float(input(""))
    TT = 2*pi*sqrt(m/k)
    print(TT)
period()

def culc():
    oper = int(input())
    x = float(input(""))
    y = float(input(""))
    print(x+y)
    if (oper == 1):
        print(x+y)
    if (oper == 2):
        print(x-y)
    if (oper == 3):
        print(x*y)
    if (oper == 4):
        if (x!=0):
            print(x**y)
    if (oper == 5):
        if (y!=0):
            print(x/y)
culc()
